os与strings常用函数

os包中的一些常用函数

fmt.Println(os.Hostname())//获取主机名

fmt.Println(os.Getwd());//获取当前目录

fmt.Println(os.Getuid());//获取用户ID

fmt.Println(os.Geteuid());//获取有效用户ID

fmt.Println(os.Getgid());//获取组ID

fmt.Println(os.Getegid());//获取有效组ID

fmt.Println(os.Getpid());//获取进程ID

fmt.Println(os.Getppid());//获取父进程ID

fmt.Println(os.Getenv("GOPATH"));//获取环境变量的值

os.Setenv("TEST", "test");//设置环境变量的值

//获取所有环境变量

fmt.Println(strings.Join(os.Environ(), “\r\n”));

//把字符串中带${var}或$var替换成指定指符串

//创建目录
os.Mkdir(“abc”, os.ModePerm);

//创建多级目录
os.MkdirAll(“abc/d/e/f”, os.ModePerm);

//删除文件或目录
os.Remove(“abc/d/e/f”);

//删除指定目录下所有文件
os.RemoveAll(“abc”);

//重命名文件
os.Rename(“./2.txt”, “./2_new.txt”);

//判断是否为同一文件,unix下通过底层结构的设备和索引节点是否相同来判断,其他系统可能是通过文件绝对路径来判断

//返回临时目录
fmt.Println(os.TempDir());
